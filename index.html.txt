<!DOCTYPE html>
<html>
<head>
    <title>Homework Helper</title>
</head>
<body>
    <h1>Homework Helper</h1>
    <p>Type your question or upload a file/photo below:</p>
    <form>
        <label for="question">Type your question:</label><br>
        <textarea id="question" name="question" rows="4" cols="50"></textarea><br><br>
        <label for="file">Or upload a file/photo:</label>
        <input type="file" id="file" name="file"><br><br>
        <input type="submit" value="Submit">
    </form>
    <div id="output"></div>
    <img id="imagePreview" src="" style="max-width:300px; display:none; margin-top:20px;" alt="Image preview will appear here." />
    <div id="fileName" style="margin-top:10px;"></div>
    <script>
    document.querySelector('form').addEventListener('submit', function(event) {
        event.preventDefault(); // Prevent page reload
        const question = document.getElementById('question').value;
        const outputDiv = document.getElementById('output');
        if (question.trim() !== "") {
            outputDiv.innerHTML = "<h2>Your Question:</h2><p>" + question + "</p>";
            // Simulated step-by-step solution
            outputDiv.innerHTML += `
                <h3>Step-by-Step Solution:</h3>
                <ol>
                    <li>Read the question carefully.</li>
                    <li>Identify what is being asked.</li>
                    <li>List out known information and requirements.</li>
                    <li>Apply relevant formulas or logic.</li>
                    <li>Write out the answer clearly.</li>
                </ol>
                <p><em>(This is a placeholder. You can connect to a real AI or backend later!)</em></p>
            `;
        } else {
            outputDiv.innerHTML = "<p>Please enter a question or upload a file.</p>";
        }
    });

    // Image and file name preview functionality
    document.getElementById('file').addEventListener('change', function(event) {
        const fileInput = event.target;
        const imagePreview = document.getElementById('imagePreview');
        const fileNameDiv = document.getElementById('fileName');
        const file = fileInput.files[0];

        if (file) {
            fileNameDiv.textContent = "Uploaded file: " + file.name;
        } else {
            fileNameDiv.textContent = "";
        }

        if (file && file.type.startsWith('image/')) {
            const reader = new FileReader();
            reader.onload = function(e) {
                imagePreview.src = e.target.result;
                imagePreview.style.display = "block";
            }
            reader.readAsDataURL(file);
        } else {
            imagePreview.src = "";
            imagePreview.style.display = "none";
        }
    });
    </script>
</body>
</html>
